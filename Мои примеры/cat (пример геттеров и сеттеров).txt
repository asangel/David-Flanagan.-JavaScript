
!function () {

    // локальные переменные, доступные только для методов деступа(т.е. геттеров, сеттеров)

    var age = 0;
    var name = 'Barsik';

        },
        set age(newAge) {
            age = newAge;
        }
    }

    this.cat = cat;

}();
