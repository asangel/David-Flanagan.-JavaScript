

function foreach(a, f, t) {
    try { a.forEach(f, t); }
    catch (e) {
        if (e === foreach.break) return undefined;
        else throw e;
    }
}

foreach.break = new Error("StopIteration");


var summ=0;
arr3 = [1,2,3,4,5,6,7,8];

/* вызов foreach */
foreach(arr3, function (item) {
    if(item >= 6) throw foreach.break;
    summ += item;

});


console.dir(summ);
